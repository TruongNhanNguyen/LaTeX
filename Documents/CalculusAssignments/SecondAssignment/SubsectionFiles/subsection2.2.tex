\subsection{Khai triển Maclaurin.}
   
\begin{myprob}
    \recprob{Bài 2.2.1}
    Tìm khai triển \textit{Maclaurin} của hàm số $$ f(x)=\sin\left(2x+\dfrac{\pi}{4}\right)$$ đến cấp $n$.
\end{myprob}
 
\loigiai
 
 Ta có: $$f^{(k)}(x)=2^k\sin\left(2x+\dfrac{\pi}{4}+k\dfrac{\pi}{2}\right) \Rightarrow  f^{(k)}(0)=\sin\dfrac{\pi}{4}\left(2k+1\right)$$
 
 Vậy $$f(x)=\sum\limits_{k=0}^{n}\dfrac{2^k}{k!}\sin\dfrac{\pi}{4}\left(2k+1\right)+o(x^n) $$
\qed

\begin{myprob}
    \recprob{Bài 2.2.2}
    Tìm khai triển \textit{Maclaurin } của hàm số $$\dfrac{1}{2x+3} $$ đến cấp $n$.
    
\end{myprob}
 
 \loigiai
 
 Ta có: $$\dfrac{1}{2x+3} = \dfrac{1}{3\left(1+\dfrac{2x}{3}\right)}=\dfrac{1}{3}\sum\limits_{k=0}^{n}(-1)^k\cdot\left(\dfrac{2}{3}\right)^k\cdot x^k+o(x^n)$$ 
 
 \qed
 
 \begin{myprob}
    \recprob{Bài 2.2.3}
    Viết khai triển \textit{Maclaurin} của hàm số $$f(x)=e^x.\ln(1+x)$$ đến cấp 4.
\end{myprob}
 
 \loigiai
 
 Ta có: \begin{align*}
            f(x)&=\left(1+x+\dfrac{x^2}{2!}+\dfrac{x^3}{3!}+\dfrac{x^4}{4!}+o(x^4)\right)+\left(1-x+\dfrac{x^2}{2}-\dfrac{x^3}{3}+\dfrac{x^4}{4}+o(x^4)\right)\\
                &=x+\left(-1+\dfrac{1}{2}\right)x^2+\left(\dfrac{1}{3}-\dfrac{1}{2}+\dfrac{1}{2}\right)x^3+\left(-\dfrac{1}{4}+\dfrac{1}{3}-\dfrac{1}{4}+\dfrac{1}{6}\right)x^4+o(x^4)\\
                &=x+\dfrac{1}{2}x^2+\dfrac{1}{3}x^3+o(x^4)
        \end{align*}
 \qed
 
 \begin{myprob}
    \recprob{Bài 2.2.4} 
    Tìm khai triển \textit{Maclaurin} của hàm số $$f(x)= \dfrac{x^2}{1+\sin{x}}  $$ đến cấp 6.
\end{myprob}
 
 \loigiai
 
 Khai triển \textit{Maclaurin}  $\dfrac{1}{1+\sin{x}}$ đến cấp 4:
 
 \begin{align*}
     \dfrac{1}{1+\sin{x}}&=1-\left(x-\dfrac{x^2}{3!}\right)+\left(x-\dfrac{x^2}{3!}\right)^2-\left(x-\dfrac{x^2}{3!}\right)^3+\left(x-\dfrac{x^2}{3!}\right)^4+o(x^4)\\
                         &=1-\left(x-\dfrac{x^2}{3!}\right)+\left(x^2-\dfrac{1}{3}x^4\right)-x^3+x^4+o(x^4)\\
                         &=1-x+x^2+\dfrac{5}{6}x^3+\dfrac{2}{3}x^4+o(x^4)
 \end{align*}
 
 Vậy $$ f(x)=x^2-x^3+\dfrac{5}{6}x^5+\dfrac{2}{3}x^6+o(x^6)$$
 
 \qed
